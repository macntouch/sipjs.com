In order to make and receive calls and messages you must create a new SIP user agent. You do so by doing this.

 1// phone.js
 2
 3function () {
 4  //Creates the anonymous user agent so that you can make calls
 5  var userAgent = new SIP.UA(config);
 6
 7  var answerButton = document.getElementById('answerButton');
 8  var endButton = document.getElementById('endButton');
 9
10  userAgent.start();
11
12  userAgent.on('start', newSessionHandler);
13}
14
15
16function newSessionHandler()
17{
18  //here you determine whether the call has video and audio
19  userAgent.on('connected', function () {
20    var options = {
21      mediaConstraints: {
22        audio: true,
23        video: true
24      } 
25    }
26    //makes the call
27    var session = userAgent.invite('sip:bob@example.com', options);
28    session.on('accepted', onAccept)
29  })
30}
31
32function onAccept () {
33  //gets the video elements
34  var remoteVideo = document.getElementById('remoteVideo');
35  var localVideo = document.getElementById('localVideo');
36
37  //attached the received video stream to the Video Elements
38  remoteVideo.srcObject= session.getRemoteStreams()[0];
39  localVideo.srcObject= session.getLocalStreams()[0];
40
41  //plays the Video Elements
42  remoteVideo.play();
43  localVideo.play();
44}
